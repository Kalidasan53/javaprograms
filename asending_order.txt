package array;

public class asending_order {

	public static void main(String[] args) {
		int []a= {3,2,4,8,1,54};
		int temp;
		for(int i=0;i<a.length;i++) {
			for(int j=i+1;j<a.length;j++) {
				if(a[i]>a[j]) {
					temp=a[i];
					a[i]=a[j];
					a[j]=temp;
				}
			}
		}
		System.out.println("asending order:-");
      for(int i=0;i<a.length;i++) {
    	  System.out.print(a[i]+" ");
      }
	}

}
