package array;

public class greater_leader {

	public static void main(String[] args) {
		int a[]= {9,8,6,5};
		boolean b1=false;
	
		for(int i=0;i<a.length;i++) 
		{
			int leader=0;
			
			for(int j=i+1;j<a.length;j++)
			{
				if(a[i]>a[j]) {
				      leader=a[i];
				      b1=true;
				     break;
				
			  }
				
			}
			if(leader>0 && b1) {
				System.out.print(leader+ " ");
			}
			
	}
if(b1==false) {
	System.out.println(" not");
}
}}
