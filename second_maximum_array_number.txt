
package array;

public class second_maximum {

	public static void main(String[] args) {
		
       int[]a= {10,3,4,5,-9,6};
       int max=a[0];
       int second_max=a[1];
       for(int i=1;i<a.length;i++) {
    	   if(a[i]>max) {
    		    		max =a[i];
    	        }
    	   else if(a[i]>second_max) {
    		   second_max=a[i];
    	   }}
       
System.out.println("first minimum is;"+max);
System.out.println("second minimum is;"+second_max);

}}
