package array;

public class second_minimum {

	public static void main(String[] args) {
		
       int[]a= {10,3,4,5,-9,6};
       int min=a[0];
       int second_min=a[1];
       for(int i=1;i<a.length;i++) {
    	   if(a[i]<min) {
    		second_min=min;
    		min =a[i];
    		
    	   }
       }
System.out.println("first minimum is;"+min);
System.out.println("second minimum is;"+second_min);
}

}
