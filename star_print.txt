package pattern_printing;

import java.util.Iterator;

public class star_print {

	public static void main(String[] args) {
		int n=4;
		for(int i=1;i<=4;i++) {
			
			for(int j=i;j<4;j++) {
				System.out.print(" ");
			}
			for(int k=1;k<=(2*i)-1;k++) {
				System.out.print("*");
			}
			System.out.println();
		}
	}

}
